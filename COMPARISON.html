from langgraph.graph import StateGraph, END

def build_graph():
    g = StateGraph(BlockingAgentState)

    g.add_node("detection",    detection_node)
    g.add_node("analyzer",     analyzer_node)
    g.add_node("determination",determination_node)
    g.add_node("action",       action_node)
    g.add_node("rca",          rca_node)
    g.add_node("notification", notification_node)

    g.set_entry_point("detection")

    g.add_conditional_edges(
        "detection", route_after_detection,
        {"analyzer": "analyzer",
         "notification": "notification"}
    )
    g.add_edge("analyzer", "determination")
    g.add_conditional_edges(
        "determination", route_after_determination,
        {"action": "action", "rca": "rca"}
    )
    g.add_edge("action", "rca")
    g.add_edge("rca", "notification")
    g.add_edge("notification", END)

    return g.compile()

AGENT_GRAPH = build_graph()
MS AGENT FRAMEWORK src/orchestrator/workflow.py
from agent_framework import (
    WorkflowBuilder, FanOutEdgeGroup
)

def build_workflow():
    builder = WorkflowBuilder(
        start_executor=detection_node
    )

    builder.add_edge_group(FanOutEdgeGroup(
        source_id="detection",
        target_ids=["analyzer", "notification"],
        selection_func=route_after_detection,
    ))
    builder.add_edge(analyzer_node, determination_node)

    builder.add_edge_group(FanOutEdgeGroup(
        source_id="determination",
        target_ids=["action", "rca"],
        selection_func=route_after_determination,
    ))
    builder.add_edge(action_node, rca_node)
    builder.add_edge(rca_node, notification_node)

    return builder.build()

AGENT_WORKFLOW = build_workflow()

LLM Client agents/base_agent.py

The most impactful change. LangGraph uses ChatOpenAI (any provider, sync). MAF uses FoundryChatClient (Azure Foundry, async). This forces all agent methods to become async.

LANGGRAPH — ChatOpenAI (synchronous)
from langchain_openai import ChatOpenAI
from langchain_core.messages import (
    HumanMessage, SystemMessage
)

def _make_llm() -> ChatOpenAI:
    return ChatOpenAI(
        model=os.getenv("LLM_MODEL", "gpt-4o"),
        api_key=os.getenv("LLM_API_KEY"),
        base_url=os.getenv("LLM_BASE_URL"),
        temperature=0,
        max_tokens=4096,
        timeout=int(os.getenv("LLM_TIMEOUT_SECONDS","30")),

            HumanMessage(content=user_message),
        ])
        return response.content  # ← synchronous

from agent_framework.foundry import FoundryChatClient
from azure.identity import DefaultAzureCredential

    return FoundryChatClient(
        project_endpoint=os.getenv(
            "FOUNDRY_PROJECT_ENDPOINT"),
        model=os.getenv("LLM_MODEL", "gpt-4.1"),

    _CLIENT = None  # lazy singleton

Node / Executor Definition

LangGraph nodes are plain Python functions. MAF executors use a decorator and must be async.

LANGGRAPH — plain function
from langchain_core.runnables import RunnableConfig

def analyzer_node(
    state: dict,
    config: RunnableConfig
) -> dict:
    cfg = config["configurable"]  # injected by framework
    agent = AnalyzerAgent(cfg)
    updates = agent.run(state)      # synchronous
    return updates                   # partial dict


# Router function returns a string
def route_after_detection(state) -> str:
    return "analyzer" if state.get("has_blocking") \
           else "notification"
MS AGENT FRAMEWORK — @executor decorator
from agent_framework import executor, WorkflowContext
from orchestrator.config import get_config

@executor(id="analyzer")
async def analyzer_node(
    state: BlockingState,
    context: WorkflowContext
) -> None:
    cfg = get_config()         # module singleton
    updates = await AnalyzerAgent(cfg).run(state)
    state.llm_analysis  = updates["llm_analysis"]
    state.severity_hint = updates["severity_hint"]
    await context.send_message(state)  # forward


# Router returns list[str] (supports true fan-out)
def route_after_detection(state) -> list[str]:
    return ["analyzer"] if state.has_blocking \
           else ["notification"]

Async/Sync Bridge MAF-only addition

The shared tools/rca.py calls ask_llm_json as a synchronous callable. In the MAF version that method is a coroutine — passing it directly would return a coroutine object, not a dict. A lightweight bridge solves this without changing the shared tool.

LANGGRAPH — tools/rca.py just works

    input: RCAInput,
    ask_llm_json: Callable[[str, str], dict],
) -> RCAOutput:
    # Called synchronously -- works fine with
    # LangGraph's synchronous ask_llm_json()
    raw = ask_llm_json(_SYSTEM_PROMPT, user_msg)
    ...

# rca/agent.py
result = generate_rca(
    rca_input,
    ask_llm_json=agent.ask_llm_json  # sync ✓
)
MS AGENT FRAMEWORK — sync bridge needed
# base_agent.py — sync wrapper for tool use
def ask_llm_json_sync(self, system, user) -> dict:
    # Run async coroutine in a fresh thread loop
    # (avoids "nested event loop" error)
    with ThreadPoolExecutor(max_workers=1) as pool:
        future = pool.submit(
            asyncio.run,
            self.ask_llm_json(system, user),
        )
        return future.result()

# rca/agent.py — use sync bridge
result = generate_rca(
    rca_input,
    ask_llm_json=agent.ask_llm_json_sync  # ✓
)

Why ThreadPoolExecutor + asyncio.run? Calling asyncio.get_event_loop().run_until_complete() inside an already-running async context raises "This event loop is already running." Running in a dedicated thread gives it a fresh loop. No extra dependencies required (no nest_asyncio).

Verdict — When to Use Each

Both implementations deliver identical blocking-detection behavior. The choice comes down to your organization's cloud and toolchain strategy.

Choose LangGraph when…

  • You need multi-provider LLM flexibility (OpenAI, RouteLLM, Abacus AI, Ollama, Anthropic)
  • You are not an Azure shop or want no Azure dependency
  • You prefer open-source with a large community and extensive docs
  • You need complex graph features: cycles, human-in-the-loop, LangGraph Studio
  • Your team already uses the LangChain ecosystem
  • You want per-invocation config injection (multi-tenant scenarios)
  • You prefer synchronous code patterns

Choose MS Agent Framework when…

  • You are an Azure-first organization (Managed Identity, RBAC, Key Vault)
  • You use Azure AI Foundry for model governance and deployment
  • You want Semantic Kernel + AutoGen capabilities in one unified SDK
  • You need enterprise-grade observability via Azure Monitor / App Insights
  • You are building in a Microsoft 365 / Copilot Studio context
  • True parallel fan-out (multiple targets per superstep) is needed
  • You are a Microsoft shop — this project exactly fits that profile
For this blocking agent specifically

Since you are a Microsoft shop using Azure AI Foundry, the MAF version is the better long-term strategic choice. It integrates natively with your existing Azure authentication (no API key rotation), uses Azure Monitor for observability, and aligns with Microsoft's platform roadmap combining Semantic Kernel + AutoGen. The LangGraph version remains the pragmatic choice for local development, multi-provider testing, or if you need to switch models to non-Azure providers quickly.
